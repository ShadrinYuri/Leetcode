#include <iostream>
#include <string>
#include <map>

using namespace::std;

int main() {
  string s, c;
  cin >> s >> c;
  map<char, int> ms, mc;

  for(auto& i : s) {
    ms[i]++;
  }

  for(auto& i : c) {
    mc[i]++;
  }

  if (ms == mc) {
    cout << 1 << endl;
  } else {
    cout << 0 << endl;
  }
  return 0;
}
